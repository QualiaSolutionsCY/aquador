**Create aria-labels.ts:**
```typescript
/**
 * ARIA label generators for accessible 3D elements
 *
 * Following W3C WAI-ARIA 1.2 Canvas Element Accessibility guidelines:
 * - Canvas should have role="img" for static scenes or role="application" for interactive
 * - Descriptive aria-label explaining what the canvas contains
 * - Live regions for dynamic state changes
 */

/**
 * Generates ARIA label for 3D product scene
 *
 * @param productName - Name of the product being displayed
 * @param isInteractive - Whether user can interact with the scene
 * @returns Descriptive ARIA label string
 */
export function get3DSceneLabel(
  productName: string,
  isInteractive = true
): string {
  const interactivity = isInteractive
    ? 'Use arrow keys to rotate, plus and minus to zoom, R to reset view.'
    : 'Static 3D visualization.';

  return `${productName} - 3D perfume bottle visualization. ${interactivity}`;
}

/**
 * Generates state announcement for screen readers
 *
 * @param action - Action performed (zoom, rotate, reset)
 * @param value - Optional value (zoom level, rotation angle)
 * @returns Announcement string for aria-live region
 */
export function get3DStateAnnouncement(
  action: 'zoom' | 'rotate' | 'reset',
  value?: number
): string {
  switch (action) {
    case 'zoom':
      return value
        ? `Zoomed to ${Math.round(value * 100)}%`
        : 'Zoom changed';
    case 'rotate':
      return 'View rotated';
    case 'reset':
      return 'View reset to default';
    default:
      return '';
  }
}

/**
 * Generates ARIA label for animated filter
 *
 * @param filterName - Name of filter being applied
 * @param count - Number of results after filtering
 * @returns Descriptive ARIA label
 */
export function getFilterLabel(filterName: string, count: number): string {
  return `${filterName} filter applied. Showing ${count} products.`;
}

/**
 * Generates ARIA label for parallax decorative element
 *
 * @param description - What the element visually represents
 * @returns ARIA label or empty string for presentation role
 */
export function getParallaxLabel(description?: string): string {
  // Decorative parallax elements should use role="presentation"
  // and not have aria-label unless they convey meaning
  return description || '';
}

/**
 * High contrast media query detection
 *
 * Returns true if user has enabled forced-colors (Windows High Contrast Mode)
 * Used to add borders and outlines for better visibility
 */
export function isHighContrastMode(): boolean {
  if (typeof window === 'undefined') return false;
  return window.matchMedia('(forced-colors: active)').matches;
}
```
