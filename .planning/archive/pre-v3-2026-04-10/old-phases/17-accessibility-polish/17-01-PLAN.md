Create keyboard navigation system for 3D OrbitControls following Three.js accessibility patterns.

**Create useKeyboardControls.ts:**
```typescript
'use client';

import { useEffect, RefObject } from 'react';
import { OrbitControls as OrbitControlsImpl } from 'three-stdlib';

export interface KeyboardControlsOptions {
  enabled?: boolean;
  rotateSpeed?: number;
  zoomSpeed?: number;
  /** Called when user rotates view via keyboard */
  onRotate?: () => void;
  /** Called when user zooms via keyboard, receives current zoom scale */
  onZoom?: (level: number) => void;
  /** Called when user resets view via keyboard */
  onReset?: () => void;
}

/**
 * Adds keyboard navigation to OrbitControls
 *
 * Keyboard shortcuts:
 * - Arrow keys: Rotate camera around target
 * - +/=: Zoom in
 * - -/_: Zoom out
 * - R: Reset camera to default position
 *
 * @param controlsRef - Ref to OrbitControls instance
 * @param options - Configuration options including optional state-change callbacks
 */
export function useKeyboardControls(
  controlsRef: RefObject<OrbitControlsImpl | null>,
  options: KeyboardControlsOptions = {}
) {
  const {
    enabled = true,
    rotateSpeed = 0.1,
    zoomSpeed = 0.1,
    onRotate,
    onZoom,
    onReset,
  } = options;

  useEffect(() => {
    if (!enabled || !controlsRef.current) return;

    const controls = controlsRef.current;
    const handleKeyDown = (e: KeyboardEvent) => {
      // Ignore if user is typing in an input
      if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) {
        return;
      }

      switch (e.key) {
        case 'ArrowLeft':
          e.preventDefault();
          controls.rotateLeft(rotateSpeed);
          controls.update();
          onRotate?.();
          break;
        case 'ArrowRight':
          e.preventDefault();
          controls.rotateLeft(-rotateSpeed);
          controls.update();
          onRotate?.();
          break;
        case 'ArrowUp':
          e.preventDefault();
          controls.rotateUp(rotateSpeed);
          controls.update();
          onRotate?.();
          break;
        case 'ArrowDown':
          e.preventDefault();
          controls.rotateUp(-rotateSpeed);
          controls.update();
          onRotate?.();
          break;
        case '+':
        case '=':
          e.preventDefault();
          controls.dollyIn(1 + zoomSpeed);
          controls.update();
          onZoom?.((controls as any).getDistance?.() ?? 1);
          break;
        case '-':
        case '_':
          e.preventDefault();
          controls.dollyOut(1 + zoomSpeed);
          controls.update();
          onZoom?.((controls as any).getDistance?.() ?? 1);
          break;
        case 'r':
        case 'R':
          e.preventDefault();
          controls.reset();
          controls.update();
          onReset?.();
          break;
      }
    };

    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [controlsRef, enabled, rotateSpeed, zoomSpeed, onRotate, onZoom, onReset]);
}
```

```typescript
'use client';

import { useState, useEffect } from 'react';
import { useReducedMotion } from '@/hooks/useReducedMotion';

interface Shortcut {
  keys: string[];
  description: string;
}

const SHORTCUTS: Shortcut[] = [
  { keys: ['←', '→', '↑', '↓'], description: 'Rotate' },
  { keys: ['+', '-'], description: 'Zoom' },
  { keys: ['R'], description: 'Reset' },
];

/**
 * Keyboard shortcuts hint overlay for 3D scenes
 *
 * Shows on first load, dismissible with X or Escape key.
 * Respects prefers-reduced-motion for fade animations.
 * Uses Phase 10 luxury design (gold accents, backdrop blur).
 */
export function KeyboardHints() {
  const [visible, setVisible] = useState(false);
  const [dismissed, setDismissed] = useState(false);
  const reducedMotion = useReducedMotion();

  useEffect(() => {
    // Show hints after 1 second if not previously dismissed
    const hasSeenHints = localStorage.getItem('3d-keyboard-hints-seen');
    if (!hasSeenHints) {
      const timer = setTimeout(() => setVisible(true), 1000);
      return () => clearTimeout(timer);
    }
  }, []);

  const handleDismiss = () => {
    setVisible(false);
    setDismissed(true);
    localStorage.setItem('3d-keyboard-hints-seen', 'true');
  };

  useEffect(() => {
    const handleEscape = (e: KeyboardEvent) => {
      if (e.key === 'Escape' && visible) {
        handleDismiss();
      }
    };
    window.addEventListener('keydown', handleEscape);
    return () => window.removeEventListener('keydown', handleEscape);
  }, [visible]);

  if (dismissed || !visible) return null;

  return (
    <div
      className={`fixed bottom-4 right-4 bg-neutral-900/90 backdrop-blur-md rounded-lg p-4 shadow-lg border border-gold-500/20 max-w-xs z-50 ${
        reducedMotion ? '' : 'animate-fadeIn'
      }`}
      role="dialog"
      aria-label="Keyboard shortcuts"
    >
      <div className="flex items-start justify-between mb-2">
        <h3 className="text-sm font-semibold text-gold-500">
          Keyboard Controls
        </h3>
        <button
          onClick={handleDismiss}
          className="text-neutral-400 hover:text-gold-500 transition-colors"
          aria-label="Dismiss keyboard hints"
        >
          ✕
        </button>
      </div>

      <ul className="space-y-2">
        {SHORTCUTS.map((shortcut, idx) => (
          <li key={idx} className="flex items-center justify-between text-xs">
            <div className="flex gap-1">
              {shortcut.keys.map((key, i) => (
                <kbd
                  key={i}
                  className="px-2 py-1 bg-neutral-800 border border-neutral-700 rounded text-neutral-300 font-mono"
                >
                  {key}
                </kbd>
              ))}
            </div>
            <span className="text-neutral-400">{shortcut.description}</span>
          </li>
        ))}
      </ul>
    </div>
  );
}
```
