```typescript
'use client';

import React, { Component, ReactNode } from 'react';
import * as Sentry from '@sentry/nextjs';

interface Props {
  children: ReactNode;
  fallback?: ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

/**
 * Real React Error Boundary
 * Catches component render errors and shows fallback UI instead of white screen
 */
export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    // Log to Sentry
    Sentry.captureException(error, {
      contexts: {
        react: {
          componentStack: errorInfo.componentStack,
        },
      },
    });

    console.error('React Error Boundary caught error:', error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      // Show custom fallback or default error UI
      if (this.props.fallback) {
        return this.props.fallback;
      }

      return (
        <div className="min-h-screen flex items-center justify-center bg-gray-950 px-4">
          <div className="max-w-md w-full bg-gray-900/50 backdrop-blur-xl border border-gold/20 rounded-2xl p-8 text-center">
            <h1 className="text-2xl font-playfair text-gold mb-4">
              Something went wrong
            </h1>
            <p className="text-gray-400 mb-6">
              We've been notified and are looking into it.
            </p>
            <button
              onClick={() => window.location.reload()}
              className="px-6 py-3 bg-gold text-black font-semibold rounded-lg hover:bg-gold/90 transition-colors"
            >
              Reload Page
            </button>
          </div>
        </div>
      );
    }

    return this.props.children;
  }
}

/**
 * Suppresses harmless AbortErrors during navigation
 * Keep this separate from ErrorBoundary — different concern
 */
export function AbortErrorSuppressor() {
  React.useEffect(() => {
    const handleUnhandledRejection = (event: PromiseRejectionEvent) => {
      if (
        event.reason?.name === 'AbortError' ||
        event.reason?.message?.includes('signal is aborted') ||
        event.reason?.message?.includes('aborted')
      ) {
        event.preventDefault();
        return;
      }
    };

    window.addEventListener('unhandledrejection', handleUnhandledRejection);
    return () => {
      window.removeEventListener('unhandledrejection', handleUnhandledRejection);
    };
  }, []);

  return null;
}
```

- Add real class-based error boundary with componentDidCatch
- Show branded fallback UI instead of white screen
- Log errors to Sentry with component stack
- Keep AbortErrorSuppressor as separate export (still needed for navigation)

3. **Validate on load** (replace lines 79-90):
   ```typescript
   useEffect(() => {
     try {
       const stored = localStorage.getItem(CART_STORAGE_KEY);
       if (stored) {
         const parsed = JSON.parse(stored);

         // Validate cart structure with Zod
         const validated = CartSchema.safeParse(parsed);

         if (validated.success) {
           dispatch({ type: 'HYDRATE', payload: validated.data });
         } else {
           // Invalid cart data — log and clear
           Sentry.captureMessage('Invalid cart data in localStorage', {
             level: 'warning',
             extra: {
               errors: validated.error.errors,
               rawData: parsed
             },
           });
           localStorage.removeItem(CART_STORAGE_KEY);
         }
       }
     } catch (e) {
       Sentry.captureException(e, {
         tags: { context: 'cart-hydration' }
       });
       localStorage.removeItem(CART_STORAGE_KEY);
     } finally {
       setIsHydrated(true);
     }
   }, []);
   ```

This ensures malformed cart data (from browser extensions, manual edits, or old schema) is cleared instead of crashing the app.
