  itemCount: items.length.toString(),

    productId: i.productId,
    variantId: i.variantId,
    name: i.name,           // REMOVE - redundant

**Optimized metadata** (essential data only):
```typescript
metadata: {
  itemCount: items.length.toString(),
  items: JSON.stringify(items.map(i => ({
    pid: i.productId,      // Shortened key
    vid: i.variantId,      // Shortened key
    qty: i.quantity,       // Shortened key
  }))),
}
```

Rationale:
- `name` is redundant (webhook can look up from productId)
- `price` is redundant (already in Stripe line_items)
- `productType` is redundant (encoded in variantId)
- Shortened keys save ~40% space per item
- Webhook will be updated in Phase 2 to handle this format

Add comment above metadata:
```typescript

// pid=productId, vid=variantId, qty=quantity
// Webhook reconstructs full data from these identifiers (SEC-04)
```

const abortControllerRef = useRef<AbortController | null>(null);
const [isProcessing, setIsProcessing] = useState(false);

const handleCheckout = async () => {
  // Prevent double-submission
  if (isProcessing) return;

  // Cancel any in-flight request
  if (abortControllerRef.current) {
    abortControllerRef.current.abort();
  }

  setIsProcessing(true);
  setIsLoading(true);
  setError(null);

  // Create new abort controller for this request
  const abortController = new AbortController();
  abortControllerRef.current = abortController;

  try {
    const response = await fetch('/api/checkout', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ items: cart.items }),
      signal: abortController.signal, // Add abort signal
    });

    // ... existing response handling ...

  } catch (error) {
    // Handle abort separately
    if (error instanceof Error && error.name === 'AbortError') {
      console.log('Checkout request cancelled');
      return;
    }
    // ... existing error handling ...
  } finally {
    setIsProcessing(false);
    setIsLoading(false);
    abortControllerRef.current = null;
  }
};
