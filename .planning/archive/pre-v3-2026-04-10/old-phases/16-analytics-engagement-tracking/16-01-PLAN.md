   **Pattern A — OrbitControls ref (preferred if no ref exists yet):**
   Add a ref to the OrbitControls element:
   ```typescript
   import { OrbitControls } from '@react-three/drei';
   import type { OrbitControls as OrbitControlsImpl } from 'three-stdlib';

   const controlsRef = useRef<OrbitControlsImpl>(null);

   // In JSX:
   <OrbitControls ref={controlsRef} {...ORBIT_CONFIG} />

   // In useEffect:
   useEffect(() => {
     const controls = controlsRef.current;
     if (!controls) return;

     const handleStart = () => {
       startTimeRef.current = Date.now();
       track3DInteraction('rotate_start', { productName });
     };

     const handleEnd = () => {
       const duration = Date.now() - startTimeRef.current;
       track3DInteraction('rotate_end', { productName, duration });
     };

     controls.addEventListener('start', handleStart);
     controls.addEventListener('end', handleEnd);

     return () => {
       controls.removeEventListener('start', handleStart);
       controls.removeEventListener('end', handleEnd);
     };
   }, [productName]);
   ```

   ```typescript
     const duration = Date.now() - startTimeRef.current;
     track3DInteraction('rotate_end', { productName, duration });
   };

   // In JSX:
   <OrbitControls
     {...ORBIT_CONFIG}
     onStart={handleControlStart}
     onEnd={handleControlEnd}
   />
   ```

**Implementation pattern for scroll depth:**
```typescript
export function useScrollDepthTracking(pageName: string) {
  const { scrollYProgress } = useScroll();

  useEffect(() => {
    const storageKey = `scroll_depth_${pageName}`;
    const tracked = sessionStorage.getItem(storageKey)?.split(',') || [];

    return scrollYProgress.on('change', (progress) => {
      const percentage = Math.floor(progress * 100);

      [25, 50, 75, 100].forEach(milestone => {
        if (percentage >= milestone && !tracked.includes(String(milestone))) {
          trackScrollDepth(milestone, pageName);
          tracked.push(String(milestone));
          sessionStorage.setItem(storageKey, tracked.join(','));
        }
      });
    });
  }, [pageName, scrollYProgress]);
}
```

**Do NOT:**
- Track every scroll position (only milestones)
- Fire events on every scroll frame (use milestones)
- Track if user quickly scrolls past (IntersectionObserver handles this)
